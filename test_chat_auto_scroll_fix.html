Copy and paste this script into your browser console to start monitoring chat scroll behavior:
// Chat Auto-Scroll Diagnostic Script console.log('🔍 [CHAT SCROLL TEST] Starting enhanced diagnostic monitoring...'); function startChatScrollTest() { const messagesContainer = document.querySelector('[class*="overflow-y-auto"]'); const messagesEnd = document.querySelector('div[class*="messagesEnd"]'); if (!messagesContainer) { console.error('❌ [TEST] Messages container not found. Make sure you are on the Exploration Map page.'); return null; } console.log('✅ [TEST] Found messages container:', messagesContainer); let testResults = { initialLoad: false, streamingScroll: false, newMessageScroll: false, userScrollUp: false, userScrollDown: false }; const logScrollState = (event) => { const { scrollTop, scrollHeight, clientHeight } = messagesContainer; const distanceFromBottom = scrollHeight - scrollTop - clientHeight; const isAtBottom = distanceFromBottom < 100; console.log(`📜 [TEST-${event}] Scroll state:`, { scrollTop, scrollHeight, clientHeight, distanceFromBottom, isAtBottom, timestamp: new Date().toLocaleTimeString() }); return isAtBottom; }; // Monitor for auto-scroll during content changes const observer = new MutationObserver((mutations) => { mutations.forEach((mutation) => { if (mutation.type === 'childList' && mutation.addedNodes.length > 0) { console.log('🆕 [TEST] New content detected, checking auto-scroll...'); setTimeout(() => { const isAtBottom = logScrollState('CONTENT-ADDED'); if (isAtBottom) { console.log('✅ [TEST] Auto-scroll working - user is at bottom after content addition'); testResults.newMessageScroll = true; } else { console.log('⚠️ [TEST] Auto-scroll may not be working - user not at bottom after content addition'); } }, 200); } }); }); observer.observe(messagesContainer, { childList: true, subtree: true }); // Initial state check logScrollState('INITIAL'); return { testResults, logScrollState, stop: () => { observer.disconnect(); console.log('🛑 [TEST] Monitoring stopped.'); } }; } // Start the test window.chatScrollTest = startChatScrollTest(); console.log('🎮 [TEST] Use window.chatScrollTest.stop() to stop monitoring');
